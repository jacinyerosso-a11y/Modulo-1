# 1) Instalar Postfix + herramientas de correo
sudo apt update
sudo apt install postfix mailutils -y

# Durante el asistente:
# - General type of mail configuration: Internet Site
# - System mail name: mailserver (o tu dominio si tienes)

# 2) Asegurar que Postfix esté corriendo
sudo systemctl enable postfix
sudo systemctl restart postfix
sudo systemctl status postfix --no-pager

# 3) Configuración básica (asegura envío local y salida SMTP)
sudo postconf -e "myhostname = mailserver"
sudo postconf -e "myorigin = /etc/mailname"
sudo postconf -e "inet_interfaces = all"
sudo postconf -e "inet_protocols = all"
sudo postconf -e "mydestination = \$myhostname, localhost.localdomain, localhost"
sudo postconf -e "mynetworks = 127.0.0.0/8 [::1]/128"
sudo postconf -e "home_mailbox = Maildir/"

# 4) Aplicar cambios y recargar
sudo systemctl restart postfix

# 5) Abrir puerto 25 (si usas UFW; si no tienes UFW, ignora)
sudo ufw allow 25/tcp 2>/dev/null || true
sudo ufw reload 2>/dev/null || true

# 6) Probar conectividad a internet (si esto falla, NO saldrá el correo)
ping -c 3 8.8.8.8

# 7) Enviar el correo EXACTO al maestro
echo "Jacinyer Rosso 2025-1440" | mail -s "MambruSeFueALaGuerra" os3conadrian@gmail.com

# 8) Ver si quedó en cola (debe quedar vacío o bajar rápido)
mailq

# 9) Ver el log del intento de envío (lo que el profe ama)
sudo tail -n 80 /var/log/mail.log
